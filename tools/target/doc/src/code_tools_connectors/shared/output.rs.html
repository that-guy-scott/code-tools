4/// Output format options for CLI commands
5#[derive(Debug, Clone, Copy)]
6pub enum OutputFormat {
7    Json,
8    Text,
9    Csv,
10}
11
12impl fmt::Display for OutputFormat {
13    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
14        match self {
15            OutputFormat::Json => write!(f, "json"),
16            OutputFormat::Text => write!(f, "text"),
17            OutputFormat::Csv => write!(f, "csv"),
18        }
19    }
20}
21
22impl std::str::FromStr for OutputFormat {
23    type Err = anyhow::Error;
24    
25    fn from_str(s: &str) -> Result<Self, Self::Err> {
26        match s.to_lowercase().as_str() {
27            "json" => Ok(OutputFormat::Json),
28            "text" => Ok(OutputFormat::Text),
29            "csv" => Ok(OutputFormat::Csv),
30            _ => Err(anyhow::anyhow!("Invalid output format: {}", s)),
31        }
32    }
33}

35/// Format output data based on requested format
36pub fn format_output(data: &Value, format: OutputFormat) -> String {
37    match format {
38        OutputFormat::Json => serde_json::to_string_pretty(data).unwrap_or_default(),
39        OutputFormat::Text => format_as_text(data),
40        OutputFormat::Csv => format_as_csv(data),
41    }
42}
43
44/// Format JSON value as human-readable text
45fn format_as_text(data: &Value) -> String {
46    match data {
47        Value::Array(arr) => {
48            arr.iter()
49                .enumerate()
50                .map(|(i, item)| format!("{}. {}", i + 1, format_value_as_text(item)))
51                .collect::<Vec<_>>()
52                .join("\n")
53        }
54        Value::Object(obj) => {
55            obj.iter()
56                .map(|(key, value)| format!("{}: {}", key, format_value_as_text(value)))
57                .collect::<Vec<_>>()
58                .join("\n")
59        }
60        _ => format_value_as_text(data),
61    }
62}
63
64/// Format a single JSON value as text
65fn format_value_as_text(value: &Value) -> String {
66    match value {
67        Value::String(s) => s.clone(),
68        Value::Number(n) => n.to_string(),
69        Value::Bool(b) => b.to_string(),
70        Value::Null => "null".to_string(),
71        Value::Array(arr) => format!("[{}]", arr.iter().map(format_value_as_text).collect::<Vec<_>>().join(", ")),
72        Value::Object(obj) => {
73            let fields: Vec<String> = obj.iter()
74                .map(|(k, v)| format!("{}: {}", k, format_value_as_text(v)))
75                .collect();
76            format!("{{{}}}", fields.join(", "))
77        }
78    }
79}

81/// Format JSON data as CSV (if it's an array of objects)
82fn format_as_csv(data: &Value) -> String {
83    match data {
84        Value::Array(arr) if !arr.is_empty() => {
85            // Check if all elements are objects with the same keys
86            if let Some(Value::Object(first_obj)) = arr.first() {
87                let mut headers: Vec<String> = first_obj.keys().cloned().collect();
88                headers.sort();
89                let header_line = headers.join(",");
90                
91                let rows: Vec<String> = arr.iter()
92                    .filter_map(|item| {
93                        if let Value::Object(obj) = item {
94                            let row: Vec<String> = headers.iter()
95                                .map(|header| {
96                                    let value = obj.get(header)
97                                        .map(format_csv_value)
98                                        .unwrap_or_default();
99                                    escape_csv_value(&value)
100                                })
101                                .collect();
102                            Some(row.join(","))
103                        } else {
104                            None
105                        }
106                    })
107                    .collect();
108                
109                format!("{}\n{}", header_line, rows.join("\n"))
110            } else {
111                // Fallback to text format for non-object arrays
112                format_as_text(data)
113            }
114        }
115        _ => format_as_text(data),
116    }
117}
118
119/// Format a JSON value for CSV output
120fn format_csv_value(value: &Value) -> String {
121    match value {
122        Value::String(s) => s.clone(),
123        Value::Number(n) => n.to_string(),
124        Value::Bool(b) => b.to_string(),
125        Value::Null => String::new(),
126        _ => serde_json::to_string(value).unwrap_or_default(),
127    }
128}
129
130/// Escape a CSV value (wrap in quotes if contains comma, quote, or newline)
131fn escape_csv_value(value: &str) -> String {
132    if value.contains(',') || value.contains('"') || value.contains('\n') {
133        format!("\"{}\"", value.replace('"', "\"\""))
134    } else {
135        value.to_string()
136    }
137}

139#[cfg(test)]
140mod tests {
141    use super::*;
142    use serde_json::json;

144    #[test]
145    fn test_output_format_parsing() {
146        assert!(matches!("json".parse::<OutputFormat>().unwrap(), OutputFormat::Json));
147        assert!(matches!("text".parse::<OutputFormat>().unwrap(), OutputFormat::Text));
148        assert!(matches!("csv".parse::<OutputFormat>().unwrap(), OutputFormat::Csv));
149        assert!("invalid".parse::<OutputFormat>().is_err());
150    }
151
152    #[test]
153    fn test_json_formatting() {
154        let data = json!({"name": "test", "value": 42});
155        let formatted = format_output(&data, OutputFormat::Json);
156        assert!(formatted.contains("\"name\": \"test\""));
157    }
158
159    #[test]
160    fn test_text_formatting() {
161        let data = json!({"name": "test", "value": 42});
162        let formatted = format_output(&data, OutputFormat::Text);
163        assert!(formatted.contains("name: test"));
164        assert!(formatted.contains("value: 42"));
165    }
166
167    #[test]
168    fn test_csv_formatting() {
169        let data = json!([
170            {"name": "Alice", "age": 30},
171            {"name": "Bob", "age": 25}
172        ]);
173        let formatted = format_output(&data, OutputFormat::Csv);
174        assert!(formatted.starts_with("age,name"));
175        assert!(formatted.contains("30,Alice"));
176        assert!(formatted.contains("25,Bob"));
177    }
178
179    #[test]
180    fn test_csv_escaping() {
181        assert_eq!(escape_csv_value("simple"), "simple");
182        assert_eq!(escape_csv_value("with,comma"), "\"with,comma\"");
183        assert_eq!(escape_csv_value("with\"quote"), "\"with\"\"quote\"");
184    }
